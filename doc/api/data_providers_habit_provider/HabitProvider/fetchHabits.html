Future<void> fetchHabits()

Fetches all habits from the repository and initializes categories for those habits. Notifies listeners when the data is loaded or an error occurs.

Implementation

Future<void> fetchHabits() async {
  _isLoading = true;
  notifyListeners();

  try {
    _habits = await _habitRepository.getAllHabits();
    categoryProvider.initializeCategories(_habits);
  } catch (e) {
    debugPrint('Error fetching habits: $e');
  } finally {
    _isLoading = false;
    notifyListeners();
  }
}
